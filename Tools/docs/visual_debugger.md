```python
def regression_test(baseline_dir, current_dir):
    """–°—Ä–∞–≤–Ω–µ–Ω–∏–µ —Å —ç—Ç–∞–ª–æ–Ω–Ω—ã–º–∏ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è–º–∏"""
    
    from PIL import Image, ImageChops
    
    baseline_images = Path(baseline_dir).glob('*.png')
    
    for baseline_path in baseline_images:
        current_path = Path(current_dir) / baseline_path.name
        
        if not current_path.exists():
            print(f"‚ùå –û—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç: {baseline_path.name}")
            continue
            
        # –°—Ä–∞–≤–Ω–µ–Ω–∏–µ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π
        baseline_img = Image.open(baseline_path)
        current_img = Image.open(current_path)
        
        diff = ImageChops.difference(baseline_img, current_img)
        
        if diff.getbbox():
            print(f"‚ö†Ô∏è –†–∞–∑–ª–∏—á–∏—è –≤: {baseline_path.name}")
        else:
            print(f"‚úÖ –ò–¥–µ–Ω—Ç–∏—á–Ω–æ: {baseline_path.name}")
```
